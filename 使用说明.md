user root;
master_process off;
daemon off;
error_log /tmp/nginx.log debug;
events {
    worker_connections  1024;
}
rtmp {
    server {
        listen 1935;
        chunk_size 4000;
        application hls {
            rgw_endpoint http://10.139.11.165:8000;
            live on;
            hls on;
            hls_path /usr/local/nginx/html/;
            hls_fragment 10s;
            hls_playlist_length 50s;
            hls_cleanup on;
        }
    }
}
http {
    server {
        listen  8080;
        location /hls {
            # Serve HLS fragments
            types {
                application/vnd.apple.mpegurl m3u8;
                video/mp2t ts;
            }
            root html;
            expires -1;
        }
    }
}
